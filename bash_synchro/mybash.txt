#!/bin/bash
#host variables 
HOST1="192.168.56.10" 
#paths variables 
PATHM1=~/backup/hostdata1 
PATHLOGS=~/backup 
CHECKPATH=$( basename $1 ) 
CHECKDIR=$( dirname $1 )
PATHHOST1=$CHECKDIR/$CHECKPATH
#massives 
hostsmas=( "$HOST1" ) 
foldermymas=( "$PATHM1" ) 
#cycle for multi copying 
for (( i=0; i<1; i++ )) 
do 
	if [[ `ssh ${hostsmas[$i]} test -d $PATHHOST1 && echo exists` ]] 
	then 
		#check what file we need to copy
		LISTTAR=""
		LISTTMP=$( rsh ${hostsmas[$i]} find $PATHHOST1/ -type f )
		LISTTO=""
		PATHHOSTTMP=$PATHHOST1/
		for inv in $LISTTMP
		do
			TMPL=${inv#*$PATHHOSTTMP}
			LISTTO="$TMPL $LISTTO"
		done
		for name in $LISTTO
		do
			if [ -e ${foldermymas[$i]}/$CHECKPATH/$name ]
			then
				SUMMCH1=$( rsh ${hostsmas[$i]} md5sum $PATHHOST1/$name | cut -f1 -d' ' )
				SUMMCH2=$( md5sum ${foldermymas[$i]}/$CHECKPATH/$name | cut -f1 -d' ' )
				if [ $SUMMCH1 != $SUMMCH2 ]
				then
					LISTTAR="$name $LISTTAR"
				fi
			else
				LISTTAR="$name $LISTTAR"
			fi
		done
		#check the bug about empty archive
		if [ ! -z $LISTTAR ]
		then
			rsh ${hostsmas[$i]} tar cfP $PATHHOST1\.tar -C $PATHHOST1 $LISTTAR 
			#starting timer for 5 minutes allowed in background 
			~/timer5 & 
			#Actually process of copying 
			scp -pr ${hostsmas[$i]}:$PATHHOST1\.tar ${foldermymas[$i]} 
			SUMHOST1=$( rsh ${hostsmas[$i]} md5sum $PATHHOST1\.tar | cut -f1 -d' ' ) 
			SUMM1=$( md5sum ${foldermymas[$i]}/$CHECKPATH\.tar | cut -f1 -d' ' )
		fi 
	else 
		~/timer5 & 
		scp -pr ${hostsmas[$i]}:$PATHHOST1 ${foldermymas[$i]} 
		SUMHOST1=$( rsh ${hostsmas[$i]} md5sum $PATHHOST1 | cut -f1 -d' ' ) 
		SUMM1=$( md5sum ${foldermymas[$i]}/$CHECKPATH | cut -f1 -d' ' ) 
	fi 
	#Date of end of the copying 
	DATA=$(date) 
	#Checking control sums of copied files 
	if [ "$SUMHOST1" = "$SUMM1" ] 
	then 
		echo "$DATA	Host ${hostsmas[$i]} : copying done!" >> $PATHLOGS/log 
		if [[ `ssh ${hostsmas[$i]} test -d $PATHHOST1 && echo exists` ]] 
		then 
			if[ ! -z $LISTTAR ]
			then
				#create emprty folders (for full synchro)
				if [ ! -d ${foldermymas[$i]}/$CHECKPATH ] 
				then 
					mkdir ${foldermymas[$i]}/$CHECKPATH 
				fi 
				LISTTMP=$( rsh ${hostsmas[$i]} find $PATHHOST1/* -type d )
				LISTTO=""
				PATHHOSTTMP=$PATHHOST1/
				for inv in $LISTTMP
				do
					TMPL=${inv#*$PATHHOSTTMP}
					LISTTO="$TMPL $LISTTO"
				done
				for name in $LISTTO
				do
					if [ $name != $PATHHOST1 ]
					then
						if [ ! -d ${foldermymas[$i]}/$CHECKPATH/$name ]
						then
							mkdir -p ${foldermymas[$i]}/$CHECKPATH/$name
						fi
					fi
				done
				tar xfP ${foldermymas[$i]}/$CHECKPATH\.tar -C ${foldermymas[$i]}/$CHECKPATH 
				rsh ${hostsmas[$i]} rm $PATHHOST1\.tar 
				rm ${foldermymas[$i]}/$CHECKPATH\.tar
			fi 
		fi 
	else 
		echo "$DATA	Host ${hostsmas[$i]} : copying done with error: some copied files are corrupted or do not exist!" >> $PATHLOGS/log 
		if [[ `ssh ${hostsmas[$i]} test -d $PATHHOST1 && echo exists` ]] 
		then 
			rsh ${hostsmas[$i]} rm $PATHHOST1\.tar 
			rm ${foldermymas[$i]}/$CHECKPATH\.tar 
		else 
			rm ${foldermymas[$i]}/$CHECKPATH 
		fi 
	fi 
done
